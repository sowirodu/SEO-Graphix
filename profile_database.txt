<?php
// Check if the form is submitted
if ($_SERVER["REQUEST_METHOD"] == "POST") {
  // Retrieve the form data
  $name = $_POST["name"];
  $email = $_POST["email"];
  $location = $_POST["location"];
  $college = $_POST["college"];
  $major = $_POST["major"];
  $career = $_POST["career"];
  $dreamCompany = $_POST["dreamcompany"];
  $linkedin = $_POST["linkedin"];
  $projects = $_POST["projects"];

  // Validate the form data (you can add your own validation logic here)

  // Connect to the database
  $servername = "localhost";
  $username = "your_username";
  $password = "your_password";
  $dbname = "your_database_name";

  $conn = new mysqli($servername, $username, $password, $dbname);

  // Check the connection
  if ($conn->connect_error) {
    die("Connection failed: " . $conn->connect_error);
  }

  // Prepare and execute the SQL statement
  $sql = "INSERT INTO profiles (name, email, location, college, major, career, dream_company, linkedin, projects) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)";
  $stmt = $conn->prepare($sql);
  $stmt->bind_param("sssssssss", $name, $email, $location, $college, $major, $career, $dreamCompany, $linkedin, $projects);
  $stmt->execute();

  // Check if the data is inserted successfully
  if ($stmt->affected_rows > 0) {
    echo "Profile created successfully!";
  } else {
    echo "Error creating profile.";
  }

  // Close the database connection
  $stmt->close();
  $conn->close();
}
?>
